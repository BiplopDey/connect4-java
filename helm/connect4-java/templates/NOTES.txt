1. Get the application URL by running these commands:

{{- if contains "LoadBalancer" .Values.service.type }}
  export SERVICE_IP=$(kubectl get svc {{ include "connect4-java.fullname" . }} -o jsonpath='{.status.loadBalancer.ingress[0].ip}')
  echo "URL: http://$SERVICE_IP:{{ .Values.service.port }}"
{{- else if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "connect4-java.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo "URL: http://$NODE_IP:$NODE_PORT"
{{- else }}
  kubectl port-forward service/{{ include "connect4-java.fullname" . }} 8080:{{ .Values.service.port }}
{{- end }}
